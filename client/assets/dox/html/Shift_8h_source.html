282 DoubleInteger<T2> Shift::LogicalRightShift(DoubleInteger<T2> number,
283  uint8 shift) {
284 
285  if (static_cast<T2>(-1) < 0) {
286 
287  // shift of sizeof(T)*8 is treated as shift 0
288  // for this reason exit here to avoid this pitfall
289  if (shift > 0u) {
290  T2 lower = number.GetLower();
291  T2 upper = number.GetUpper();
292 
293  // calculates n of bits of T
294  uint16 bitSize = static_cast<uint16>(sizeof(T2) * 8u);
295  // shift within one half
296  if (shift < bitSize) {
297  // shift lower first
298  lower = lower >> shift;
299  // add overflow from upper
300  // this would fail if shift is 0
301  lower |= (upper << (bitSize - shift));
302  // complete the upper
303  upper = upper >> shift;
304  }
305  else { // more than half!
306  // remove half
307  shift -= bitSize;
308  // swap upper -> lower and shift with the remainder
309  lower = upper >> shift;
310  // upper is 0
311  if (upper < static_cast<T2>(0)) {
312  upper = static_cast<T2>(-1);
313  }
314  else {
315  upper = static_cast<T2>(0);
316  }
317  }
318  number.SetLower(lower);
319  number.SetUpper(upper);
320  }
321  }
322  else {
323  number >>= shift;
324  }
