316 void DoubleInteger<T>::operator<<=(uint16 shift) {
317  // shift of sizeof(T)*8 is treated as shift 0
318  // for this reason exit here to avoid this pitfall
319  if (shift > 0u) {
320 
321  // calculates n of bits of T
322  uint16 bitSize = static_cast<uint16>(sizeof(T) * 8u);
323  // shift within one half
324  if (shift < bitSize) {
325  // shift upper first
326  upper = upper << shift;
327  // add overflow from lower
328  // this would fail if shift is 0
329  upper |= (lower >> (bitSize - shift));
330  // complete the lower
331  lower = lower << shift;
332  }
333  else { // more than half!
334  // remove half
335  shift -= bitSize;
336  // swap lower -> upper and shift with the remainder
337  upper = lower << shift;
